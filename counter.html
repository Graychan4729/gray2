<body>

    <div id="counter-container">

        <button onclick="decrement()">-</button>

        <div id="counter">0</div>

        <button onclick="increment()">+</button>

    </div>


    <script>

        // Load counter value from local storage or start at 0

        let counterValue = localStorage.getItem('counterValue') || 0;

        document.getElementById('counter').innerText = counterValue;


        function increment() {

            counterValue++;

            updateCounter();

            showAnimation("💗");

        }


        function decrement() {

            counterValue--;

            updateCounter();

            showAnimation("💔");

        }


        function updateCounter() {

            document.getElementById('counter').innerText = counterValue;

            localStorage.setItem('counterValue', counterValue); // Save to local storage

        }


        function showAnimation(symbol) {

            const animationElement = document.createElement("div");

            animationElement.className = "animation-icon";

            animationElement.innerText = symbol;

            document.getElementById("counter-container").appendChild(animationElement);


            // 1秒后删除动画元素

            setTimeout(() => {

                animationElement.remove();

            }, 1000);

        }

    </script>

</body>
